Inference API

To get started hit the /enrich endpoints (example below) /enrich?src=https://hips.hearstapps.com/hmg-prod/images/dog-puppy-on-garden-royalty-free-image-1586966191.jpg&mode=classify&model=resnet 
const axios = require('axios');
let data = JSON.stringify({
    "src": [
        "https://hips.hearstapps.com/hmg-prod/images/dog-puppy-on-garden-royalty-free-image-1586966191.jpg",
        "https://hips.hearstapps.com/hmg-prod/images/dog-puppy-on-garden-royalty-free-image-1586966191.jpg",
        "https://hips.hearstapps.com/hmg-prod/images/dog-puppy-on-garden-royalty-free-image-1586966191.jpg"
        ],
    "mode": "classify",
    "format": "image"
    });
                
let config = {
    method: 'post',
    maxBodyLength: Infinity,
    url: 'localhost:5000/enrich',
    headers: { 
    'Content-Type': 'application/json'
    },
    data : data
};

axios.request(config)
.then((response) => {
    console.log(JSON.stringify(response.data));
})
.catch((error) => {
    console.log(error);
});
